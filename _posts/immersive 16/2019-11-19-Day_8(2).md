## 1.1. Destructuring: Object
```javascript
ES5 Syntax

var user = {
    first: 'Benji',
    vast: 'Marinacci',
    age: 32
};

var first = user.first;
var last = user.last;

first;  // 'Benji'
```

```javascript
ES6 Syntax

var user = {
    first: 'Benji',
    vast: 'Marinacci',
    age: 32
};

var {first, last} = user;
first;  // 'Benji'
```


## 1.2. Destructuring: Array
```javascript
ES5 Syntax

var numbers = [1, 2, 3, 4];

var indexZero = numbers[0];
var indexOne = numbers[1];
var indexTwo = numbers[2];

indexZero;  // 1
```

```javascript
ES6 Syntax

var numbers = [1, 2, 3, 4];

var [indexZero, indexOne, indexTwo] = numbers;

indexZero;  // 1
```

* 함수의 파라미터에서 사용

```javascript
ES5 Syntax

var countToTwo = function(numbers) {
    var indexZero = number[0];
    var indexOne = number[1];

    console.log(indexZero + ', ' + indexOne);
};

countToTwo(numbers);    // 1, 2
```

```javascript
ES6 Syntax

var countToTwo = function([indexZero, indexOne]) {
    console.log(indexZero + ', ' + indexOne);
};

countToTwo(numbers);    // 1, 2
```
